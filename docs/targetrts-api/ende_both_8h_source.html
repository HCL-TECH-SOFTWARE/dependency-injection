352/*----------------------------------------------------------------------------
353| Decode the buffer back to a decent structure.
354+-----------------------------------------------------------------------------
355| If decoding was successful:
356| returns DEC_SUCCESS and sets buf->status to same as return value
357| sets buf->used to number of bytes decoded
358| sets info->func to the enum code describing *ptr
359| sets info->ptr to point to a suitable struct. This struct is malloc'ed
360| by TO_decodeBuffer. YOU are responsible for freeing it, and a good
361| help is to use TO_freeStructInfo().
362|
363| If decoding was unsuccessful:
364| returns != DEC_SUCCESS (see enum decodingResult above for details),
365| and sets buf->status to same as return value
366| sets buf->used to _roughly_ where in buf->data something went wrong
367| sets info->func to FUNC_NONE
368| sets info->ptr to nullptr
369| No dynamic structures are allocated in this case.
370+-----------------------------------------------------------------------------
371| Example:
372|
373| TO_bufferInfo buf;
374| TO_structInfo info;
375|
376| TO_evtPresence myPresence;
377|
378| // YOU put decodeable data into buf.data and buf.size
379| // You don't need to set buf.used, buf.status or any info.* fields
380|
381| TO_decodeBuffer( &buf, &info );
382| switch( buf.status )
383| {
384| case DEC_SUCCESS:
385| switch( info.func )
386| {
387| case EVT_PRESENCE: myPresence = *(TO_evtPresence *)info.ptr;
388| // Use the presence information in some way...
389| break;
390| case ...
391| default: // I can't handle this kind of function.
392| }
393| TO_freeStructInfo( &info );
394| break;
395| case DEC_BAD_PARM:
396| // One or both parameters to TO_decodeBuffer were null pointers.
397| break;
398| case DEC_NO_STR_LEN:
399| // No string length field at or around buf.data[ buf.used ].
400| break;
401| case ... // Other error codes here.
402| }
403+---------------------------------------------------------------------------*/
404int TO_decodeBuffer( TO_bufferInfo * buf, TO_structInfo * info );

406/*----------------------------------------------------------------------------
407| Free whatever is in the info structure.
408+-----------------------------------------------------------------------------
409| This routine assumes that the field info->func actually matches the
410| data type info->ptr points to. Manual fiddling here can be disastrous.
411+---------------------------------------------------------------------------*/
412void TO_freeStructInfo( TO_structInfo * info );
